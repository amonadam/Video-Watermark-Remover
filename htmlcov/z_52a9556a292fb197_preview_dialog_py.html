248 def numpy_to_qimage(self, image): 

249 """将numpy数组转换为QImage""" 

250 if image is None: 

251 return QImage() 

252 

253 if len(image.shape) == 2: 

254 # 灰度图像 

255 h, w = image.shape 

256 bytes_per_line = w 

257 return QImage(image.data, w, h, bytes_per_line, QImage.Format_Grayscale8) 

258 else: 

259 # 彩色图像 

260 h, w, ch = image.shape 

261 bytes_per_line = ch * w 

262 

263 # 确保是RGB格式 

264 if ch == 3: 

265 # BGR转RGB 

266 rgb_image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB) 

267 return QImage(rgb_image.data, w, h, bytes_per_line, QImage.Format_RGB888) 

268 elif ch == 4: 

269 # BGRA转RGBA 

270 rgba_image = cv2.cvtColor(image, cv2.COLOR_BGRA2RGBA) 

271 return QImage(rgba_image.data, w, h, bytes_per_line, QImage.Format_RGBA8888) 

272 else: 

273 # 其他格式转为RGB 

274 rgb_image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB) 

275 return QImage(rgb_image.data, w, h, bytes_per_line, QImage.Format_RGB888)
